text_files = [
    'email1.txt',

# Initialize a dictionary to store the counts for each email in each text file
email_counts_in_text_files = {file: {} for file in text_files}

# Function to count emails in a file
def count_emails_in_file(file_path):
    email_counts = {}
    with open(file_path, 'r') as file:
        lines = file.readlines()
        for line in lines:
            # Assuming each line contains an email and we are checking the format
            # You can adjust this based on your file format
            for email in email_counts_in_csv:  # Check against the emails in the CSV file
                if email in line:
                    if email in email_counts:
                        email_counts[email] += 1
                    else:
                        email_counts[email] = 1
    return email_counts

# Step 3: Process each text file and count occurrences of emails
for file in text_files:
    email_counts_in_text_files[file] = count_emails_in_file(file)

# Step 4: Print the counts of each email in the text files and compare with the CSV counts
for file, email_counts in email_counts_in_text_files.items():
    print(f"\nEmail counts in {file}:")
    for email, count in email_counts.items():
        csv_count = email_counts_in_csv.get(email, 0)  # Get count from CSV or 0 if not found
        print(f"{email}: {count} (CSV count: {csv_count})")
        if count != csv_count:
            print(f"WARNING: The count for {email} does not match between CSV and {file}")
